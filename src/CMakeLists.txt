cmake_minimum_required(VERSION 3.0)
set(EXECUTABLE_NAME "testmode")
find_package(BISON)
find_package(FLEX)
find_package(Threads)
BISON_TARGET(parser testmode.y ${CMAKE_CURRENT_BINARY_DIR}/testmode.c)
FLEX_TARGET(scanner testlex.l ${CMAKE_CURRENT_BINARY_DIR}/testlex.c)
ADD_FLEX_BISON_DEPENDENCY(scanner parser)

include_directories(${CMAKE_CURRENT_BINARY_DIR})
include_directories(${CMAKE_CURRENT_SOURCE_DIR})
if (!WIN32)
add_executable(${EXECUTABLE_NAME} ${BISON_parser_OUTPUTS} ${FLEX_scanner_OUTPUTS} slip.c serialtun.c)
endif(!WIN32)
add_executable(web_server server.cpp mongoose.c)
target_link_libraries(web_server ${CMAKE_THREAD_LIBS_INIT} )
INSTALL(TARGETS web_server DESTINATION bin)
